5   package net.sourceforge.pmd.lang.java.rule.codestyle;
6   
7   import java.util.Arrays;
8   import java.util.Collection;
9   
10  import net.sourceforge.pmd.lang.java.ast.ASTClassOrInterfaceDeclaration;
11  import net.sourceforge.pmd.lang.java.ast.ASTConstructorDeclaration;
12  import net.sourceforge.pmd.lang.java.ast.ASTFieldDeclaration;
13  import net.sourceforge.pmd.lang.java.ast.ASTMethodDeclaration;
14  import net.sourceforge.pmd.lang.java.rule.AbstractIgnoredAnnotationRule;
15  import net.sourceforge.pmd.lang.java.rule.design.UseUtilityClassRule;
16  
17  /**
18   * This rule detects non-static classes with no constructors;
19   * requiring even the default constructor to be explicit.
20   * It ignores classes with solely static methods,
21   * use {@link UseUtilityClassRule} to flag those.
22   */
23  public class AtLeastOneConstructorRule extends AbstractIgnoredAnnotationRule {
24  
25      public AtLeastOneConstructorRule() {
26          addRuleChainVisit(ASTClassOrInterfaceDeclaration.class);
27      }
28  
29      @Override
30      protected Collection<String> defaultSuppressionAnnotations() {
31          return Arrays.asList("lombok.Data",
32                  "lombok.Value",
33                  "lombok.Builder",
34                  "lombok.NoArgsConstructor",
35                  "lombok.RequiredArgsConstructor",
36                  "lombok.AllArgsConstructor");
37      }
38  
39      @Override
40      public Object visit(final ASTClassOrInterfaceDeclaration node, final Object data) {
41          // Ignore interfaces / static classes / classes that have a constructor / classes ignored through annotations
42          if (node.isInterface() || node.isStatic() || node.hasDescendantOfType(ASTConstructorDeclaration.class)
43                  || hasIgnoredAnnotation(node)) {
44              return data;
45          }
46  
47          boolean atLeastOneMember = false;
48  
49          // Do we have any non-static methods?
50          for (final ASTMethodDeclaration m : node.findDescendantsOfType(ASTMethodDeclaration.class)) {
51              if (!m.isStatic()) {
52                  addViolation(data, node);
53                  return data;
54              }
55              atLeastOneMember = true;
56          }
57  
58          // .. or fields?
59          for (final ASTFieldDeclaration f : node.findDescendantsOfType(ASTFieldDeclaration.class)) {
60              if (!f.isStatic()) {
61                  addViolation(data, node);
62                  return data;
63              }
64              atLeastOneMember = true;
65          }
66  
67          // Class has no declared members
68          if (!atLeastOneMember) {
69              addViolation(data, node);
70          }
71  
72          return data;
73      }
74  }
