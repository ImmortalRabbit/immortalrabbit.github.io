9   import org.jaxen.Context;
10  import org.jaxen.Function;

12  import org.jaxen.SimpleFunctionContext;
13  import org.jaxen.XPathFunctionContext;

15  import net.sourceforge.pmd.annotation.InternalApi;
16  import net.sourceforge.pmd.lang.ast.Node;
17  import net.sourceforge.pmd.lang.java.ast.TypeNode;
18  import net.sourceforge.pmd.lang.java.typeresolution.TypeHelper;
19  
20  
21  @InternalApi
22  @Deprecated
23  public class TypeIsExactlyFunction implements Function {
24  
25      public static void registerSelfInSimpleContext() {
26          ((SimpleFunctionContext) XPathFunctionContext.getInstance()).registerFunction(null, "typeIsExactly",
27                  new TypeIsExactlyFunction());
28      }
29  
30      @Override
31      public Object call(final Context context, final List args) throws FunctionCallException {
32          if (args.size() != 1) {
33              throw new IllegalArgumentException(
34                      "typeIsExactly function takes a single String argument with the fully qualified type name to check against.");
35          }
36          final String fullTypeName = (String) args.get(0);
37          final Node n = (Node) context.getNodeSet().get(0);
38          
39          return typeIsExactly(n, fullTypeName);
40      }
41  
42      /**
43       * Example XPath 1.0: {@code //ClassOrInterfaceType[typeIsExactly('java.lang.String')]}
44       * <p>
45       * Example XPath 2.0: {@code //ClassOrInterfaceType[pmd-java:typeIsExactly('java.lang.String')]}
46       *
47       * @param n The node on which to check for types
48       * @param fullTypeName The fully qualified name of the class or any supertype
49       * @return True if the type of the node matches, false otherwise.
50       */
51      public static boolean typeIsExactly(final Node n, final String fullTypeName) {
52          if (n instanceof TypeNode) {
53              return TypeHelper.isExactlyA((TypeNode) n, fullTypeName);
54          } else {
55              throw new IllegalArgumentException("typeIsExactly function may only be called on a TypeNode.");
56          }
57      }
58  }
