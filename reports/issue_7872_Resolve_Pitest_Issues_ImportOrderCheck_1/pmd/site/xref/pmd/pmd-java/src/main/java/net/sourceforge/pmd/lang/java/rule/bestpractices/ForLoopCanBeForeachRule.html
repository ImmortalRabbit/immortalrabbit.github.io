41      public ForLoopCanBeForeachRule() {
42          addRuleChainVisit(ASTForStatement.class);
43      }
44  
45      @Override
46      public Object visit(ASTForStatement node, Object data) {
47  
48          final ASTForInit init = node.getFirstChildOfType(ASTForInit.class);
49          final ASTForUpdate update = node.getFirstChildOfType(ASTForUpdate.class);
50          final ASTExpression guardCondition = node.getFirstChildOfType(ASTExpression.class);
51  
52          if (init == null && update == null || guardCondition == null) {
53              return data;
54          }
55  
56          Entry<VariableNameDeclaration, List<NameOccurrence>> indexDecl = getIndexVarDeclaration(init, update);
57  
58          if (indexDecl == null) {
59              return data;
60          }
61  
62  
63          List<NameOccurrence> occurrences = indexDecl.getValue();
64          VariableNameDeclaration index = indexDecl.getKey();
65  
66          if (TypeHelper.isExactlyAny(index, Iterator.class)) {
67              Entry<VariableNameDeclaration, List<NameOccurrence>> iterableInfo = getIterableDeclOfIteratorLoop(index, node.getScope());
68  
69              if (iterableInfo != null && isReplaceableIteratorLoop(indexDecl, guardCondition, iterableInfo, node)) {
70                  addViolation(data, node);
71              }
72              return data;
73          }
74  
75  
76          if (occurrences == null || !"int".equals(index.getTypeImage()) || !indexStartsAtZero(index)) {
77              return data;
78          }
79  
80  
81          String itName = index.getName();
82          String iterableName = getIterableNameOrNullToAbort(guardCondition, itName);
83  
84  
85          if (!isForUpdateSimpleEnough(update, itName) || iterableName == null) {
86              return data;
87          }
88  
89          Entry<VariableNameDeclaration, List<NameOccurrence>> iterableInfo = findDeclaration(iterableName, node.getScope());
90          VariableNameDeclaration iterableDeclaration = iterableInfo == null ? null : iterableInfo.getKey();
91  
92          if (iterableDeclaration == null) {
93              return data;
94          }
95  
96          if (iterableDeclaration.isArray() && isReplaceableArrayLoop(node, occurrences, iterableDeclaration)) {
97              addViolation(data, node);
98          } else if (iterableDeclaration.getTypeImage() != null && iterableDeclaration.getTypeImage()
99                                                                                      .matches("List|ArrayList|LinkedList")
100             && isReplaceableListLoop(node, occurrences, iterableDeclaration)) {
101             addViolation(data, node);
102         }
103 
104         return data;
105     }

108     /* Finds the declaration of the index variable and its occurrences, null to abort */
109     private Entry<VariableNameDeclaration, List<NameOccurrence>> getIndexVarDeclaration(ASTForInit init, ASTForUpdate update) {
110         if (init == null) {
111             return guessIndexVarFromUpdate(update);
112         }
113 
114         ASTLocalVariableDeclaration decl = init.getFirstChildOfType(ASTLocalVariableDeclaration.class);
115         if (decl == null) {
116             return null;
117         }
118 
119         int numDeclaredVars = decl.findChildrenOfType(ASTVariableDeclarator.class).size();
120         if (numDeclaredVars > 1) {
121             return null; // will abort in the calling function
122         }
123 
124         Map<VariableNameDeclaration, List<NameOccurrence>> decls = init.getScope().getDeclarations(VariableNameDeclaration.class);
125         Entry<VariableNameDeclaration, List<NameOccurrence>> indexVarAndOccurrences = null;
126 
127         for (Entry<VariableNameDeclaration, List<NameOccurrence>> e : decls.entrySet()) {
128 
129             ASTForInit declInit = e.getKey().getNode().getFirstParentOfType(ASTForInit.class);
130             if (Objects.equals(declInit, init)) {
131                 indexVarAndOccurrences = e;
132                 break;
133             }
134         }
135 
136         return indexVarAndOccurrences;
137 
138     }

141     /** Does a best guess to find the index variable, gives up if the update has several statements */
142     private Entry<VariableNameDeclaration, List<NameOccurrence>> guessIndexVarFromUpdate(ASTForUpdate update) {
143 
144         Node name = null;
145         try {
146             List<Node> match = update.findChildNodesWithXPath(getSimpleForUpdateXpath(null));
147             if (!match.isEmpty()) {
148                 name = match.get(0);
149             }
150         } catch (JaxenException je) {
151             throw new RuntimeException(je);
152         }
153 
154         if (name == null || name.getImage() == null) {
155             return null;
156         }
157 
158         return findDeclaration(name.getImage(), update.getScope().getParent());
159     }

162     /**
163      * @return true if there's only one update statement of the form i++ or ++i.
164      */
165     private boolean isForUpdateSimpleEnough(ASTForUpdate update, String itName) {
166         return update != null && update.hasDescendantMatchingXPath(getSimpleForUpdateXpath(itName));
167     }

170     private String getSimpleForUpdateXpath(String itName) {
171         return "./StatementExpressionList[count(*)=1]"
172             + "/StatementExpression"
173             + "/*[self::PostfixExpression and @Image='++' or self::PreIncrementExpression]"
174             + "/PrimaryExpression"
175             + "/PrimaryPrefix"
176             + "/Name"
177             + (itName == null ? "" : "[@Image='" + itName + "']");
178     }

181     /* We only report loops with int initializers starting at zero. */
182     private boolean indexStartsAtZero(VariableNameDeclaration index) {
183         ASTVariableDeclaratorId name = (ASTVariableDeclaratorId) index.getNode();
184         ASTVariableDeclarator declarator = name.getFirstParentOfType(ASTVariableDeclarator.class);
185 
186         if (declarator == null) {
187             return false;
188         }
189 
190         try {
191             List<Node> zeroLiteral = declarator.findChildNodesWithXPath(
192                 "./VariableInitializer/Expression/PrimaryExpression/PrimaryPrefix/Literal[@Image='0' and "
193                     + "@StringLiteral='false']");
194             if (!zeroLiteral.isEmpty()) {
195                 return true;
196             }
197         } catch (JaxenException je) {
198             throw new RuntimeException(je);
199         }
200 
201         return false;
202 
203     }

206     /**
207      * Gets the name of the iterable array or list.
208      *
209      * @param itName The name of the iterator variable
210      *
211      * @return The name, or null if it couldn't be found or the guard condition is not safe to refactor (then abort)
212      */
213     private String getIterableNameOrNullToAbort(ASTExpression guardCondition, String itName) {
214 
215 
216         if (guardCondition.getNumChildren() > 0
217             && guardCondition.getChild(0) instanceof ASTRelationalExpression) {
218 
219             ASTRelationalExpression relationalExpression = (ASTRelationalExpression) guardCondition.getChild(0);
220 
221             if (relationalExpression.hasImageEqualTo("<") || relationalExpression.hasImageEqualTo("<=")) {
222 
223                 try {
224                     List<Node> left = guardCondition.findChildNodesWithXPath(
225                         "./RelationalExpression/PrimaryExpression/PrimaryPrefix/Name[@Image='" + itName + "']");
226 
227                     List<Node> right = guardCondition.findChildNodesWithXPath(
228                         "./RelationalExpression[@Image='<']/PrimaryExpression/PrimaryPrefix"
229                             + "/Name[matches(@Image,'\\w+\\.(size|length)')]"
230                             + "|"
231                             + "./RelationalExpression[@Image='<=']/AdditiveExpression[count(*)=2 and "
232                             + "@Image='-' and PrimaryExpression/PrimaryPrefix/Literal[@Image='1']]"
233                             + "/PrimaryExpression/PrimaryPrefix/Name[matches(@Image,'\\w+\\.(size|length)')]");
234 
235                     if (left.isEmpty()) {
236                         return null;
237                     } else if (!right.isEmpty()) {
238                         return right.get(0).getImage().split("\\.")[0];
239                     } else {
240                         return null;
241                     }
242 
243                 } catch (JaxenException je) {
244                     throw new RuntimeException(je);
245                 }
246             }
247         }
248         return null;
249     }

252     private Entry<VariableNameDeclaration, List<NameOccurrence>> getIterableDeclOfIteratorLoop(VariableNameDeclaration indexDecl, Scope scope) {
253         Node initializer = indexDecl.getNode().getFirstParentOfType(ASTVariableDeclarator.class)
254                                     .getFirstChildOfType(ASTVariableInitializer.class);
255 
256         if (initializer == null) {
257             return null;
258         }
259 
260         ASTName nameNode = initializer.getFirstDescendantOfType(ASTName.class);
261         if (nameNode == null) {
262             // TODO : This can happen if we are calling a local / statically imported method that returns the iterable - currently unhandled
263             return null;
264         }
265 
266         String name = nameNode.getImage();
267         int dotIndex = name.indexOf('.');
268 
269         if (dotIndex > 0) {
270             name = name.substring(0, dotIndex);
271         }
272 
273         return findDeclaration(name, scope);
274     }

277     private boolean isReplaceableArrayLoop(ASTForStatement stmt, List<NameOccurrence> occurrences,
278                                            VariableNameDeclaration arrayDeclaration) {
279         String arrayName = arrayDeclaration.getName();
280 
281 
282         for (NameOccurrence occ : occurrences) {
283 
284             if (occ.getLocation().getFirstParentOfType(ASTForUpdate.class) == null
285                 && occ.getLocation().getFirstParentOfType(ASTExpression.class)
286                 != stmt.getFirstChildOfType(ASTExpression.class)
287                 && !occurenceIsArrayAccess(occ, arrayName)) {
288                 return false;
289             }
290         }
291         return true;
292     }

295     private boolean occurenceIsArrayAccess(NameOccurrence occ, String arrayName) {
296         if (occ.getLocation() instanceof ASTName) {
297             ASTPrimarySuffix suffix = occ.getLocation().getFirstParentOfType(ASTPrimarySuffix.class);
298 
299             if (suffix == null || !suffix.isArrayDereference()) {
300                 return false;
301             }
302 
303             return suffix.hasDescendantMatchingXPath("./Expression/PrimaryExpression[count(*)"
304                                                          + "=1]/PrimaryPrefix/Name[@Image='" + occ.getImage() + "']")
305                 && suffix.hasDescendantMatchingXPath("../PrimaryPrefix/Name[@Image='" + arrayName + "']")
306                 && !suffix.hasDescendantMatchingXPath("../../AssignmentOperator");
307         }
308         return false;
309     }

312     private boolean isReplaceableListLoop(ASTForStatement stmt, List<NameOccurrence> occurrences,
313                                           VariableNameDeclaration listDeclaration) {
314 
315         String listName = listDeclaration.getName();
316 
317 
318         for (NameOccurrence occ : occurrences) {
319 
320             if (occ.getLocation().getFirstParentOfType(ASTForUpdate.class) == null
321                 && occ.getLocation().getFirstParentOfType(ASTExpression.class)
322                 != stmt.getFirstChildOfType(ASTExpression.class)
323                 && !occurenceIsListGet(occ, listName)) {
324                 return false;
325             }
326         }
327 
328         return true;
329     }

332     /** @return true if this occurence is as an argument to List.get on the correct list */
333     private boolean occurenceIsListGet(NameOccurrence occ, String listName) {
334         if (occ.getLocation() instanceof ASTName) {
335             ASTPrimarySuffix suffix = occ.getLocation().getFirstParentOfType(ASTPrimarySuffix.class);
336 
337             if (suffix == null) {
338                 return false;
339             }
340 
341             Node prefix = suffix.getParent().getChild(0);
342 
343             if (!(prefix instanceof ASTPrimaryPrefix) || prefix.getNumChildren() != 1
344                 || !(prefix.getChild(0) instanceof ASTName)) {
345                 // it's either not a primary prefix, doesn't have children (can happen with this./super.)
346                 // or first child is not a name
347                 return false;
348             }
349 
350             String callImage = prefix.getChild(0).getImage();
351 
352             return (listName + ".get").equals(callImage);
353 
354         }
355         return false;
356     }

359     private Entry<VariableNameDeclaration, List<NameOccurrence>> findDeclaration(String varName, Scope innermost) {
360         Scope currentScope = innermost;
361 
362         while (currentScope != null) {
363             for (Entry<VariableNameDeclaration, List<NameOccurrence>> e : currentScope.getDeclarations(VariableNameDeclaration.class).entrySet()) {
364                 if (e.getKey().getName().equals(varName)) {
365                     return e;
366                 }
367             }
368             currentScope = currentScope.getParent();
369         }
370 
371         return null;
372     }

375     private boolean isReplaceableIteratorLoop(Entry<VariableNameDeclaration, List<NameOccurrence>> indexInfo,
376                                               ASTExpression guardCondition,
377                                               Entry<VariableNameDeclaration, List<NameOccurrence>> iterableInfo,
378                                               ASTForStatement stmt) {
379 
380         if (isIterableModifiedInsideLoop(iterableInfo, stmt)) {
381             return false;
382         }
383 
384 
385         String indexName = indexInfo.getKey().getName();
386 
387         if (indexName == null) {
388             return false;
389         }
390 
391         if (!guardCondition.hasDescendantMatchingXPath(
392             "./PrimaryExpression/PrimaryPrefix/Name[@Image='" + indexName + ".hasNext']")) {
393             return false;
394         }
395 
396         List<NameOccurrence> occurrences = indexInfo.getValue();
397 
398         if (occurrences.size() > 2) {
399             return false;
400         }
401 
402         for (NameOccurrence occ : indexInfo.getValue()) {
403             ScopedNode location = occ.getLocation();
404             boolean isCallingNext = location instanceof ASTName
405                     && (location.hasImageEqualTo(indexName + ".hasNext")
406                             || location.hasImageEqualTo(indexName + ".next"));
407 
408             if (!isCallingNext) {
409                 return false;
410             }
411         }
412         return true;
413     }
414 
415     private boolean isIterableModifiedInsideLoop(Entry<VariableNameDeclaration, List<NameOccurrence>> iterableInfo,
416                                                  ASTForStatement stmt) {
417 
418         String iterableName = iterableInfo.getKey().getName();
419         for (NameOccurrence occ : iterableInfo.getValue()) {
420             ASTForStatement forParent = occ.getLocation().getFirstParentOfType(ASTForStatement.class);
421             if (Objects.equals(forParent, stmt)) {
422                 String image = occ.getLocation().getImage();
423                 if (image.startsWith(iterableName + ".remove")) {
424                     return true;
425                 }
426             }
427         }
428 
429         return false;
430     }
