22  public class ScalaRuleTest extends BaseScalaTest {
23  
24      private static final String SCALA_TEST = "/parserFiles/helloworld.scala";
25  
26      @Test
27      public void testRuleVisits() {
28          final AtomicInteger visited = new AtomicInteger();
29          ScalaRule rule = new ScalaRule() {
30  
31              @Override
32              public RuleContext visit(ScalaNode<?> node, RuleContext data) {
33                  visited.incrementAndGet();
34                  return super.visit(node, data);
35              }
36          };
37          ASTSource root = scala.parseResource(SCALA_TEST);
38          rule.apply(Collections.singletonList(root), null);
39          Assert.assertEquals(12, visited.get());
40      }
41  
42      @Test
43      public void testDummyRule() {
44          ScalaRule rule = new ScalaRule() {
45              @Override
46              public RuleContext visit(ASTTermApply node, RuleContext data) {
47                  ASTTermName child = node.getFirstChildOfType(ASTTermName.class);
48                  if (child != null && child.hasImageEqualTo("println")) {
49                      addViolation(data, node);
50                  }
51                  return data;
52              }
53          };
54          Report report = scala.getReportForResource(rule, SCALA_TEST);
55  
56          Assert.assertEquals(1, IteratorUtil.count(report.iterator()));
57      }
