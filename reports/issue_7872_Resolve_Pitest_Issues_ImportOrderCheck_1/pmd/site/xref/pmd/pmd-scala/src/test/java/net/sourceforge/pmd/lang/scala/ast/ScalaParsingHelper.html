7   import java.io.File;
8   import java.io.StringReader;
9   
10  import net.sourceforge.pmd.PMD;
11  import net.sourceforge.pmd.PMDException;
12  import net.sourceforge.pmd.Report;
13  import net.sourceforge.pmd.Rule;
14  import net.sourceforge.pmd.RuleContext;
15  import net.sourceforge.pmd.RuleSet;
16  import net.sourceforge.pmd.RuleSetFactory;
17  import net.sourceforge.pmd.RuleSets;
18  import net.sourceforge.pmd.lang.ast.test.BaseParsingHelper;
19  import net.sourceforge.pmd.lang.scala.ScalaLanguageModule;
20  
21  public final class ScalaParsingHelper extends BaseParsingHelper<ScalaParsingHelper, ASTSource> {
22  
23      public static final ScalaParsingHelper DEFAULT = new ScalaParsingHelper(Params.getDefaultProcess());
24  
25      private ScalaParsingHelper(Params params) {
26          super(ScalaLanguageModule.NAME, ASTSource.class, params);
27      }
28  
29      @Override
30      protected ScalaParsingHelper clone(Params params) {
31          return new ScalaParsingHelper(params);
32      }
33  
34  
35      public Report getReportForTestString(Rule rule, String testSourceCode) {
36          PMD p = new PMD();
37          RuleContext ctx = new RuleContext();
38          Report report = new Report();
39          ctx.setReport(report);
40          ctx.setSourceCodeFile(new File("test.scala"));
41          RuleSet rules = new RuleSetFactory().createSingleRuleRuleSet(rule);
42          try {
43              p.getSourceCodeProcessor().processSourceCode(new StringReader(testSourceCode), new RuleSets(rules), ctx);
44          } catch (PMDException e) {
45              throw new AssertionError(e);
46          }
47          return report;
48      }
49  
50      public Report getReportForResource(Rule rule, String resourcePath) {
51          return getReportForTestString(rule, readResource(resourcePath));
52      }
53  }
