5   package net.sourceforge.pmd.cpd;
6   
7   import java.io.StringReader;
8   
9   import net.sourceforge.pmd.cpd.internal.JavaCCTokenizer;
10  import net.sourceforge.pmd.lang.TokenManager;
11  import net.sourceforge.pmd.lang.ast.GenericToken;
12  import net.sourceforge.pmd.lang.ecmascript5.Ecmascript5TokenManager;
13  import net.sourceforge.pmd.lang.ecmascript5.ast.Ecmascript5ParserConstants;
14  import net.sourceforge.pmd.lang.ecmascript5.ast.Token;
15  import net.sourceforge.pmd.util.IOUtil;
16  
17  /**
18   * The Ecmascript Tokenizer
19   */
20  public class EcmascriptTokenizer extends JavaCCTokenizer {
21  
22      @Override
23      protected TokenManager getLexerForSource(SourceCode sourceCode) {
24          StringBuilder buffer = sourceCode.getCodeBuffer();
25          return new Ecmascript5TokenManager(IOUtil.skipBOM(new StringReader(buffer.toString())));
26      }
27  
28      @Override
29      protected TokenEntry processToken(Tokens tokenEntries, GenericToken currentToken, String filename) {
30          return new TokenEntry(getTokenImage(currentToken), filename, currentToken.getBeginLine(),
31                  currentToken.getBeginColumn(), currentToken.getEndColumn());
32      }
33  
34      private String getTokenImage(GenericToken token) {
35          Token jsToken = (Token) token;
36          // Remove line continuation characters from string literals
37          if (jsToken.kind == Ecmascript5ParserConstants.STRING_LITERAL
38                  || jsToken.kind == Ecmascript5ParserConstants.UNTERMINATED_STRING_LITERAL) {
39              return token.getImage().replaceAll("(?<!\\\\)\\\\(\\r\\n|\\r|\\n)", "");
40          }
41          return token.getImage();
42      }
43  }
