14  public enum TimedOperationCategory {
15      RULE,
16      RULECHAIN_RULE,
17      COLLECT_FILES,
18      LOAD_RULES,
19      PARSER,
20      QUALIFIED_NAME_RESOLUTION,
21      SYMBOL_TABLE,
22      DFA,
23      TYPE_RESOLUTION,
24      RULECHAIN_VISIT,
25      MULTIFILE_ANALYSIS,
26      REPORTING,
27      FILE_PROCESSING,
28      UNACCOUNTED;
29      
30      public String displayName() {
31          final String[] parts = name().toLowerCase(Locale.getDefault()).split("_");
32          final StringBuilder sb = new StringBuilder();
33          for (final String part : parts) {
34              sb.append(StringUtils.capitalize(part)).append(' ');
35          }
36          sb.setLength(sb.length() - 1); // remove the final space
37          return sb.toString();
38      }
