7   import org.apache.commons.lang3.StringUtils;

9   import net.sourceforge.pmd.annotation.Experimental;

12  /**
13   * Transitional class until we move to Java 8.
14   * Used to build a validator until we move the static factory on the interface. Do not use.
15   *
16   * @author Cl√©ment Fournier
17   * @since 6.10.0
18   */
19  @Experimental
20  final class ConstraintFactory {
21  
22      private ConstraintFactory() {
23  
24      }
25  
26  
27      /**
28       * Builds a new validator from a predicate, and description.
29       *
30       * @param pred                  The predicate. If it returns
31       *                              false on a value, then the
32       *                              value is deemed to have a
33       *                              problem
34       * @param constraintDescription Description of the constraint,
35       *                              see {@link PropertyConstraint#getConstraintDescription()}.
36       * @param <U>                   Type of value to validate
37       *
38       * @return A new validator
39       */
40      @Experimental
41      public static <U> PropertyConstraint<U> fromPredicate(final Predicate<? super U> pred, final String constraintDescription) {
42          return new PropertyConstraint<U>() {
43  
44              @Override
45              public boolean test(U value) {
46                  return pred.test(value);
47              }
48  
49              // TODO message could be better, eg include name of the property
50              @Override
51              public String validate(U value) {
52                  return pred.test(value) ? null : "Constraint violated on property value '" + value + "' (" + constraintDescription + ")";
53              }
54  
55  
56              @Override
57              public String getConstraintDescription() {
58                  return StringUtils.capitalize(constraintDescription);
59              }
60  
61  
62              @Override
63              public PropertyConstraint<Iterable<? extends U>> toCollectionConstraint() {
64                  final PropertyConstraint<U> thisValidator = this;
65                  return ConstraintFactory.<Iterable<? extends U>>fromPredicate(
66                          new Predicate<Iterable<? extends U>>() {
67                              @Override
68                              public boolean test(Iterable<? extends U> us) {
69                                  for (U u : us) {
70                                      if (!thisValidator.test(u)) {
71                                          return false;
72                                      }
73                                  }
74                                  return true;
75                              }
76                          },
77                          "Components " + StringUtils.uncapitalize(thisValidator.getConstraintDescription())
78                  );
79              }
80          };
81      }
82  
83  
84      // Until we have Java 8
85      public interface Predicate<T> {
86          boolean test(T t);
87      }
88  }
