24  /**
25   * AntLogHandler sends log messages to an Ant Task, so the regular Ant logging
26   * is used.
27   *
28   * @author Wouter Zelle
29   * @deprecated Is internal API
30   */
31  @Deprecated
32  @InternalApi
33  public class AntLogHandler extends Handler {
34      private Project project;
35  
36      private static final Level DEFAULT_LEVEL = Level.WARNING;
37  
38      private static final Formatter FORMATTER = new PmdLogFormatter();
39  
40      // Maps from ant's Project.MSG_* to java.util.logging.Level
41      private static final Level[] LOG_LEVELS = {
42          Level.SEVERE,   // Project.MSG_ERR=0
43          Level.WARNING,  // Project.MSG_WARN=1
44          Level.INFO,     // Project.MSG_INFO=2
45          Level.CONFIG,   // Project.MSG_VERBOSE=3
46          Level.FINEST,   // Project.MSG_DEBUG=4
47      };
48  
49      public AntLogHandler(Project project) {
50          this.project = project;
51      }
52  
53      public Level getAntLogLevel() {
54          for (final BuildListener l : project.getBuildListeners()) {
55              Field declaredField = null;
56              try {
57                  if (l instanceof DefaultLogger) {
58                      declaredField = DefaultLogger.class.getDeclaredField("msgOutputLevel");
59                  } else if (l instanceof XmlLogger) {
60                      declaredField = XmlLogger.class.getDeclaredField("msgOutputLevel");
61                  } else if (l instanceof RecorderEntry) {
62                      declaredField = RecorderEntry.class.getDeclaredField("loglevel");
63                  } else if (l.getClass().getName().equals("org.gradle.api.internal.project.ant.AntLoggingAdapter")) {
64                      return determineGradleLogLevel(l);
65                  } else {
66                      try {
67                          declaredField = l.getClass().getDeclaredField("logLevel");
68                          if (declaredField.getType() != Integer.class && declaredField.getType() != int.class) {
69                              declaredField = null;
70                              project.log("Unsupported build listener: " + l.getClass(), Project.MSG_DEBUG);
71                          }
72                      } catch (final NoSuchFieldException e) {
73                          project.log("Unsupported build listener: " + l.getClass(), Project.MSG_DEBUG);
74                      }
75                  }
76  
77                  if (declaredField != null) {
78                      declaredField.setAccessible(true);
79                      return LOG_LEVELS[declaredField.getInt(l)];
80                  }
81  
82              } catch (final ReflectiveOperationException ignored) {
83                  // Just ignore it
84              }
85          }
86  
87          project.log("Could not determine ant log level, no supported build listeners found. "
88                  + "Log level is set to " + DEFAULT_LEVEL, Project.MSG_WARN);
89  
90          return DEFAULT_LEVEL;
91      }
92  
93      @Override
94      public void publish(LogRecord logRecord) {
95          // Map the log levels from java.util.logging to Ant
96          int antLevel;
97          Level level = logRecord.getLevel();
98          if (level == Level.FINEST) {
99              antLevel = Project.MSG_DEBUG; // Shown when -debug is supplied to
100             // Ant
101         } else if (level == Level.FINE || level == Level.FINER || level == Level.CONFIG) {
102             antLevel = Project.MSG_VERBOSE; // Shown when -verbose is supplied
103             // to Ant
104         } else if (level == Level.INFO) {
105             antLevel = Project.MSG_INFO; // Always shown
106         } else if (level == Level.WARNING) {
107             antLevel = Project.MSG_WARN; // Always shown
108         } else if (level == Level.SEVERE) {
109             antLevel = Project.MSG_ERR; // Always shown
110         } else {
111             throw new IllegalStateException("Unknown logging level"); // shouldn't
112             // get ALL
113             // or NONE
114         }
115 
116         project.log(FORMATTER.format(logRecord), antLevel);
117         if (logRecord.getThrown() != null) {
118             StringWriter stringWriter = new StringWriter();
119             PrintWriter printWriter = new PrintWriter(stringWriter, true);
120             logRecord.getThrown().printStackTrace(printWriter);
121             project.log(stringWriter.toString(), antLevel);
122         }
123     }
124 
125     @Override
126     public void close() throws SecurityException {
127         // nothing to do
128     }
129 
130     @Override
131     public void flush() {
132         // nothing to do
133     }

135     private Level determineGradleLogLevel(BuildListener l) {
136         try {
137             project.log("Detected gradle AntLoggingAdapter", Project.MSG_DEBUG);
138             Field loggerField = l.getClass().getDeclaredField("logger");
139             loggerField.setAccessible(true);
140             // org.gradle.internal.logging.slf4j.OutputEventListenerBackedLogger
141             Object logger = loggerField.get(l);
142 
143             Class<?> gradleLogLevel = l.getClass().getClassLoader().loadClass("org.gradle.api.logging.LogLevel");
144 
145             Method isLevelAtMostMethod = logger.getClass().getDeclaredMethod("isLevelAtMost", gradleLogLevel);
146             isLevelAtMostMethod.setAccessible(true);
147 
148             Object[] logLevels = gradleLogLevel.getEnumConstants();
149             // the log levels in gradle are declared in the order DEBUG, INFO, LIFECYCLE, WARN, QUIET, ERROR
150             Level[] mapping = new Level[] {
151                 Level.FINEST,   // DEBUG
152                 Level.CONFIG,   // INFO
153                 Level.INFO,     // LIFECYCLE
154                 Level.WARNING,  // WARN
155                 Level.SEVERE,   // QUIET
156                 Level.SEVERE,   // ERROR
157             };
158 
159             for (int i = 0; i < Math.min(logLevels.length, mapping.length); i++) {
160                 boolean enabled = (boolean) isLevelAtMostMethod.invoke(logger, logLevels[i]);
161                 if (enabled) {
162                     project.log("Current log level: " + logLevels[i] + " -> " + mapping[i], Project.MSG_DEBUG);
163                     return mapping[i];
164                 }
165             }
166         } catch (ReflectiveOperationException ignored) {
167             // ignored
168         }
169         project.log("Could not determine log level, falling back to default: " + DEFAULT_LEVEL, Project.MSG_WARN);
170         return DEFAULT_LEVEL;
171     }
