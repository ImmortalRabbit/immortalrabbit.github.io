5   package net.sourceforge.pmd.lang.metrics;
6   
7   import java.util.List;
8   import java.util.Map;
9   
10  import net.sourceforge.pmd.annotation.Experimental;
11  import net.sourceforge.pmd.lang.LanguageVersionHandler;
12  import net.sourceforge.pmd.lang.ast.Node;
13  import net.sourceforge.pmd.lang.ast.QualifiableNode;
14  
15  
16  /**
17   * Language-specific provider for metrics. Knows about all the metrics
18   * defined for a language. Can be used e.g. to build GUI applications
19   * like the designer, in a language independent way. Accessible through
20   * {@link LanguageVersionHandler#getLanguageMetricsProvider()}.
21   *
22   * Note: this is experimental, ie unstable until 7.0.0, after which it will probably
23   * be promoted to a real API.
24   *
25   * @param <T> Type of type declaration nodes of the language
26   * @param <O> Type of operation declaration nodes of the language
27   *
28   * @author Cl√©ment Fournier
29   * @since 6.11.0
30   */
31  @Experimental
32  public interface LanguageMetricsProvider<T extends QualifiableNode, O extends QualifiableNode> {
33  
34      /**
35       * Provides a hook to do any initializing before the first file is processed by PMD.
36       * This can be used by the metrics implementations to reset the cache.
37       */
38      void initialize();
39  
40      /**
41       * Returns a list of all supported type metric keys
42       * for the language.
43       */
44      List<? extends MetricKey<T>> getAvailableTypeMetrics();
45  
46  
47      /**
48       * Returns a list of all supported operation metric keys
49       * for the language.
50       */
51      List<? extends MetricKey<O>> getAvailableOperationMetrics();
52  
53  
54      /**
55       * Returns the given node casted to {@link T} if it's of the correct
56       * type, otherwise returns null.
57       */
58      T asTypeNode(Node anyNode);
59  
60  
61      /**
62       * Returns the given node casted to {@link O} if it's of the correct
63       * type, otherwise returns null.
64       */
65      O asOperationNode(Node anyNode);
66  
67  
68      /**
69       * Like {@link MetricsComputer#computeForType(MetricKey, QualifiableNode, boolean, MetricOptions, MetricMemoizer)},
70       * but performs no memoisation.
71       */
72      double computeForType(MetricKey<T> key, T node, MetricOptions options);
73  
74  
75      /**
76       * Like {@link MetricsComputer#computeForOperation(MetricKey, QualifiableNode, boolean, MetricOptions, MetricMemoizer)}
77       * but performs no memoisation.
78       */
79      double computeForOperation(MetricKey<O> key, O node, MetricOptions options);
80  
81  
82      /**
83       * Like {@link MetricsComputer#computeWithResultOption(MetricKey, QualifiableNode, boolean, MetricOptions, ResultOption, ProjectMemoizer)}
84       * but performs no memoisation.
85       */
86      double computeWithResultOption(MetricKey<O> key, T node, MetricOptions options, ResultOption option);
87  
88  
89      /**
90       * Computes all metrics available on the given node.
91       * The returned results may contain Double.NaN as a value.
92       *
93       * @param node Node to inspect
94       *
95       * @return A map of metric key to their result, possibly empty, but with no null value
96       */
97      Map<MetricKey<?>, Double> computeAllMetricsFor(Node node);
98  }
