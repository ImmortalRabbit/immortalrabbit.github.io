5   package net.sourceforge.pmd.util.designerbindings;
6   
7   import java.util.List;
8   
9   import net.sourceforge.pmd.annotation.Experimental;
10  import net.sourceforge.pmd.lang.ast.Node;
11  
12  /**
13   * Provides a way for the designer to highlight related nodes upon selection,
14   * eg those nodes referring to the same variable or method.
15   *
16   * <p>This API only published to bind to the designer for now, it should
17   * not be used by rule code. The criterion for selecting nodes to highlight
18   * is subject to change at the implementation's discretion. In particular it's
19   * not necessarily the usages of a variable.
20   *
21   * <p>The binary API is <b>unstable</b> until at least 7.0, meaning the only
22   * place this can be implemented safely is within PMD's own codebase.
23   *
24   * @author Cl√©ment Fournier
25   * @since 6.20.0
26   */
27  @Experimental
28  public interface RelatedNodesSelector {
29  
30  
31      /**
32       * Returns a list of nodes that should be highlighted when selecting
33       * the given node. This is eg the nodes that correspond to usages of
34       * a variable declared by the given node. If the node cannot be handled
35       * by this resolver or is otherwise uninteresting, then returns an empty list.
36       *
37       * @param node A node, with no guarantee about type. Implementations
38       *             should check their casts.
39       *
40       * @return A list of nodes to highlight
41       */
42      List<Node> getHighlightedNodesWhenSelecting(Node node);
43  
44  
45  }
