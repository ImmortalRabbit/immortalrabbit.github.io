5   package net.sourceforge.pmd.lang;
6   
7   import java.io.Reader;
8   import java.util.Map;
9   
10  import net.sourceforge.pmd.lang.ast.Node;
11  import net.sourceforge.pmd.lang.ast.ParseException;
12  
13  /**
14   * Common interface for calling tree-building parsers or source files.
15   *
16   * @author Pieter_Van_Raemdonck - Application Engineers NV/SA - www.ae.be
17   */
18  public interface Parser {
19      /**
20       * Get the ParserOptions used by this Parser.
21       */
22      ParserOptions getParserOptions();
23  
24      /**
25       * Get a TokenManager for the given source.
26       *
27       * @param fileName
28       *            The file name being parsed (may be <code>null</code>).
29       * @param source
30       *            Reader that provides the source code to tokenize.
31       * @return A TokenManager for reading token.
32       */
33      TokenManager getTokenManager(String fileName, Reader source);
34  
35  
36      /**
37       * Indicates if this parser can actual parse, or if it can only tokenize.
38       *
39       * @deprecated With 7.0.0, all parsers will be able to parse and
40       *     this method will be removed. Note that in the meantime, you
41       *     probably still need to check this method.
42       */
43      @Deprecated
44      boolean canParse();
45  
46      /**
47       * Parse source code and return the root node of the AST.
48       *
49       * @param fileName
50       *            The file name being parsed (may be <code>null</code>).
51       * @param source
52       *            Reader that provides the source code of a compilation unit
53       * @return the root node of the AST that is built from the source code
54       * @throws ParseException
55       *             In case the source code could not be parsed, probably due to
56       *             syntactical errors.
57       */
58      Node parse(String fileName, Reader source) throws ParseException;
59  
60      /**
61       * Returns the map of line numbers to suppression / review comments.
62       * Only single line comments are considered, that start with the configured
63       * "suppressMarker", which by default is "PMD". The text after the
64       * suppressMarker is used as a "review comment" and included in this map.
65       *
66       * <p>
67       * This map is later used to determine, if a violation is being suppressed.
68       * It is suppressed, if the line of the violation is contained in this suppress map.
69       *
70       * @return map of the suppress lines with the corresponding review comments.
71       *
72       * @deprecated With 7.0.0, this method will be removed. To support
73       *    suppressing with suppress markers, this method is still needed in PMD 6.
74       */
75      @Deprecated
76      Map<Integer, String> getSuppressMap();
77  }
