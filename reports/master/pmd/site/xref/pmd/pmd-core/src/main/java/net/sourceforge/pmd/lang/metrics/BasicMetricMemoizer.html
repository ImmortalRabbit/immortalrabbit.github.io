7   import java.util.HashMap;
8   import java.util.Map;

10  import net.sourceforge.pmd.lang.ast.Node;

12  /**
13   * Basic implementation of a metric memoizer.
14   *
15   * @param <N> Type of node on which the memoized metric can be computed
16   *
17   * @author Cl√©ment Fournier
18   * @since 6.0.0
19   */
20  public class BasicMetricMemoizer<N extends Node> implements MetricMemoizer<N> {
21  
22  
23      private final Map<ParameterizedMetricKey<N>, Double> memo = new HashMap<>();
24  
25  
26      @Override
27      public Double getMemo(ParameterizedMetricKey<N> key) {
28          return memo.get(key);
29      }
30  
31  
32      @Override
33      public void memoize(ParameterizedMetricKey<N> key, double value) {
34          memo.put(key, value);
35      }
36  }
