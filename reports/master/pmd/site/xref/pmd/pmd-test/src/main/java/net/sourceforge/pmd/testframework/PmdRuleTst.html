7   import java.util.Collections;
8   import java.util.List;
9   
10  import org.junit.runner.RunWith;
11  
12  import net.sourceforge.pmd.Rule;
13  
14  @RunWith(PMDTestRunner.class)
15  public class PmdRuleTst extends RuleTst {
16  
17      @Override
18      protected void setUp() {
19          // empty, nothing to do
20      }
21  
22      @Override
23      protected List<Rule> getRules() {
24          String[] packages = getClass().getPackage().getName().split("\\.");
25          String categoryName = packages[packages.length - 1];
26          String language = packages[packages.length - 3];
27          String rulesetXml = "category/" + language + "/" + categoryName + ".xml";
28  
29          Rule rule = findRule(rulesetXml, getClass().getSimpleName().replaceFirst("Test$", ""));
30          return Collections.singletonList(rule);
31      }
32  }
