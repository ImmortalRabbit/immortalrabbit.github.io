32  	// this test contains all the architectural rules that are valid for
33  	// the whole src/main/java of spoon-decompiler
34  	// we put them in the same test in order to only build the full model once
35  	// Similar to ArchitectureTest of spoon-core
36  	@Test
37  	public void testSrcMainJava() {
38  		Launcher spoon = new Launcher();
39  		spoon.getEnvironment().setCommentEnabled(true);
40  		spoon.addInputResource("src/main/java/");
41  
42  		// contract: all non-trivial public methods should be documented with proper API Javadoc
43  		spoon.buildModel();
44  		List<String> notDocumented = new ArrayList<>();
45  		for (CtMethod method : spoon.getModel().getElements(new TypeFilter<>(CtMethod.class))) {
46  
47  			// now we see whether this should be documented
48  			if (method.hasModifier(ModifierKind.PUBLIC) // public methods should be documented
49  					&& !method.getSimpleName().startsWith("get") // all kinds of setters can be undocumented
50  					&& !method.getSimpleName().startsWith("set")
51  					&& !method.getSimpleName().startsWith("is")
52  					&& !method.getSimpleName().startsWith("add")
53  					&& !method.getSimpleName().startsWith("remove")
54  					&& method.getTopDefinitions().isEmpty() // only the top declarations should be documented (not the overriding methods which are lower in the hierarchy)
55  					&& (
56  					method.hasModifier(ModifierKind.ABSTRACT) // all interface methods and abstract class methods must be documented
57  
58  							// GOOD FIRST ISSUE
59  							// ideally we want that **all** public methods are documented
60  							// so far, we have this arbitrary limit in the condition below (35)
61  							// because it's a huge task to document everything at once
62  							// so to contribute to Spoon, what you can do is
63  							// 1) you lower the threshold (eg 33)
64  							// 2) you run test `documentedTest`, it will output a list on undocumented methods
65  							// 3) you document those methods
66  							// 4) you run the test again to check that it passes
67  							// 4) you commit your changes and create the corresponding pull requests
68  							|| method.filterChildren(new TypeFilter<>(CtCodeElement.class)).list().size() > 35  // means that only large methods must be documented
69  			)) {
70  
71  				// OK it should be properly documented
72  
73  				// is it really well documented?
74  				if (method.getDocComment().length() <= 15) { // the Javadoc must be at least at least 15 characters (still pretty short...)
75  					notDocumented.add(method.getParent(CtType.class).getQualifiedName() + "#" + method.getSignature());
76  				}
77  			}
78  		}
79  		if (!notDocumented.isEmpty()) {
80  			fail(notDocumented.size() + " public methods should be documented with proper API documentation: \n" + StringUtils.join(notDocumented, "\n"));
81  		}
82  
83  		// contract: Spoon's code never uses TreeSet constructor, because they implicitly depend on Comparable (no static check, only dynamic checks)
84  		List<CtConstructorCall> treeSetWithoutComparators = spoon.getFactory().Package().getRootPackage().filterChildren(new AbstractFilter<CtConstructorCall>() {
85  			@Override
86  			public boolean matches(CtConstructorCall element) {
87  				return element.getType().getActualClass().equals(TreeSet.class) && element.getArguments().isEmpty();
88  			}
89  		}).list();
90  
91  		assertEquals(0, treeSetWithoutComparators.size());
92  	}

94  	@Test
95  	public void testGoodTestClassNames() {
96  		// contract: to be run by Maven surefire, all test classes must be called Test* or *Test
97  		// reference: "By default, the Surefire Plugin will automatically include all test classes with the following wildcard patterns:"
98  		// "**/Test*.java" and "**/*Test.java"
99  		// http://maven.apache.org/surefire/maven-surefire-plugin/examples/inclusion-exclusion.html
100 		SpoonAPI spoon = new Launcher();
101 		spoon.addInputResource("src/test/java/");
102 		spoon.buildModel();
103 
104 		for (CtMethod<?> meth : spoon.getModel().getElements(new TypeFilter<CtMethod>(CtMethod.class) {
105 			@Override
106 			public boolean matches(CtMethod element) {
107 				return super.matches(element) && element.getAnnotation(Test.class) != null;
108 			}
109 		})) {
110 			assertTrue("naming contract violated for " + meth.getParent(CtClass.class).getSimpleName(), meth.getParent(CtClass.class).getSimpleName().startsWith("Test") || meth.getParent(CtClass.class).getSimpleName().endsWith("Test"));
111 		}
112 
113 		// contract: the Spoon test suite does not depend on Junit 3 classes and methods
114 		// otherwise, intellij automatically selects the junit3 runner, finds nothing
115 		// and crashes with a dirty exception
116 		assertEquals(0, spoon.getModel().getElements(new TypeFilter<CtTypeReference>(CtTypeReference.class) {
117 			@Override
118 			public boolean matches(CtTypeReference element) {
119 				CtMethod parent = element.getParent(CtMethod.class);
120 				return "junit.framework.TestCase".equals(element.getQualifiedName());
121 			}
122 		}).size());
123 	}

126 	public void testStaticClasses() {
127 		// contract: helper classes only have static methods and a private constructor
128 
129 		SpoonAPI spoon = new Launcher();
130 		spoon.addInputResource("src/main/java/");
131 		spoon.buildModel();
132 
133 		for (CtClass<?> klass : spoon.getModel().getElements(new TypeFilter<CtClass>(CtClass.class) {
134 			@Override
135 			public boolean matches(CtClass element) {
136 				return element.getSuperclass() == null && super.matches(element) && !element.getMethods().isEmpty()
137 						&& element.getElements(new TypeFilter<>(CtMethod.class)).stream().allMatch(x -> x.hasModifier(ModifierKind.STATIC));
138 			}
139 		})) {
140 			assertTrue("Utility class " + klass.getQualifiedName() + " is missing private constructor", klass.getElements(new TypeFilter<>(CtConstructor.class)).stream().allMatch(x -> x.hasModifier(ModifierKind.PRIVATE)));
141 		}

168 	private static void assertSetEquals(String msg, Set<?> set1, Set<?> set2) {
169 		if (set1 == null || set2 == null) {
170 			throw new IllegalArgumentException();
171 		}
172 
173 		if (set1.size() != set2.size()) {
174 			throw new AssertionError(msg + "\n\nDetails: " + computeDifference(set1, set2));
175 		}
176 
177 		if (!set1.containsAll(set2)) {
178 			throw new AssertionError(msg + "\n\nDetails: " + computeDifference(set1, set2));
179 		}
180 	}
181 
182 	private static String computeDifference(Set<?> set1, Set<?> set2) {
183 		Set<String> results = new HashSet<>();
184 
185 		for (Object o : set1) {
186 			if (!set2.contains(o)) {
187 				results.add("Missing package " + o + " in computed set");
188 			} else {
189 				set2.remove(o);
190 			}
191 		}
192 
193 		for (Object o : set2) {
194 			results.add("Package " + o + " presents in computed but not expected set.");
195 		}
196 		return StringUtils.join(results, "\n");
197 	}
